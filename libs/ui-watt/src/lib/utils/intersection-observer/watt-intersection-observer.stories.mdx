# Intersection Observer

This service can be used to subscribe to changes in the intersection of a target
element with the viewport.

## Example

Use the following code to detect the first time a component becomes visible:

```ts
import { Component, ElementRef, OnInit } from '@angular/core';
import { first } from 'rxjs';
import { WattIntersectionObserverService } from '@energinet-datahub/watt/intersection-observer';

@Component({...})
class MyComponent implements OnInit {
  constructor(
    private element: ElementRef,
    private intersectionObserver: WattIntersectionObserverService
  ) {}

  ngOnInit() {
    this.intersectionObserver
      .observe(this.element.nativeElement)
      .pipe(first((entry) => entry.isIntersecting))
      .subscribe(() => console.log("NOW YOU SEE ME"));
  }
}
```
